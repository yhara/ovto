def self._run_setup(wired_action_set)
  mw = new(wired_action_set[WiredActionSet::THE_MIDDLEWARE_ITSELF])
  mw.setup
  self.middlewares.each do |m|
    m._run_setup(wired_action_set[m.name])
  end
end
